        throw e;

Сильные гарантии

  • void copyOf(CloneablePair p) {
    Cloneable temp1 = p.first.clone();
    try {
        Cloneable temp2 = p.second.clone();
        try { // Другие действия
            first = temp1; second = temp2;
        } catch (*Exception e) {
            temp2.destroy();
            throw e;
        }
    } catch (*Exception e) {
        temp1.destroy();
    }
}
